// Reads slug from URL: /embed/roof-repair-cost  OR  ?slug=roof-repair-cost
function getSlugFromURL() {
  const path = window.location.pathname.split('/')
  const idx  = path.indexOf('embed')
  if (idx >= 0 && path[idx + 1]) return path[idx + 1]
  return new URLSearchParams(window.location.search).get('slug') || 'roof-repair-cost'
}

function getOrCreateVisitorUID() {
  let uid = sessionStorage.getItem('visitor_uid')
  if (!uid) {
    uid = 'v_' + Date.now().toString(36) + Math.random().toString(36).slice(2, 8)
    sessionStorage.setItem('visitor_uid', uid)
  }
  return uid
}

async function boot() {
  try {
    // 1. Load estimator by slug
    const { data: est, error } = await sb
      .from('estimators')
      .select('id, name, slug, status, published_version_id, business_id')
      .eq('slug', SLUG)
      .eq('status', 'published')
      .single()

    if (error || !est) {
      // No published estimator — fall back to demo so you can test right away
      estimator = { id: null, name: 'Cost Estimator', slug: SLUG, business_id: null }
      steps     = getDemoSteps()
    } else {
      estimator = est

      // Load published version config
      if (est.published_version_id) {
        const { data: version } = await sb
          .from('estimator_versions')
          .select('id, config')
          .eq('id', est.published_version_id)
          .single()
        steps = version?.config?.steps || getDemoSteps()
      } else {
        steps = getDemoSteps()
      }

      // Create visitor session
      const { data: sess } = await sb.from('visitor_sessions').insert({
        visitor_uid:          visitorUID,
        estimator_id:         est.id,
        estimator_version_id: est.published_version_id || null,
        referrer:             document.referrer || null,
        user_agent:           navigator.userAgent,
        started_at:           new Date().toISOString(),
        completed:            false,
      }).select('id').single()
      sessionId = sess?.id
    }

    await fireEvent('view')
    showStep(0)

  } catch (e) {
    console.error('Boot error:', e)
    // Even on error, show demo so the widget is never blank
    estimator = { id: null, name: 'Cost Estimator', slug: SLUG, business_id: null }
    steps     = getDemoSteps()
    showStep(0)
  }

function showStep(index) {
  currentStep = index
  const step  = steps[index]
  const total = steps.length

  document.getElementById('progress-fill').style.width = (index / total * 100) + '%'
  document.getElementById('step-label').textContent    = 'Step ' + (index + 1) + ' of ' + total
  document.getElementById('step-question').textContent = step.question
  document.getElementById('step-hint').textContent     = step.hint || ''
  document.getElementById('btn-back').style.display    = index > 0 ? 'block' : 'none'

  const area = document.getElementById('step-input-area')
  if (step.type === 'choice') {
    area.innerHTML = `<div class="options">${step.options.map(o => `
      <div class="option ${answers[step.key] === o.value ? 'selected' : ''}"
           onclick="selectOption('${step.key}','${o.value}',this)">
        <div class="option-check">${answers[step.key] === o.value ? '✓' : ''}</div>
        ${o.label}
      </div>`).join('')}</div>`
  } else if (step.type === 'number') {
    const hasPfx = !!step.prefix
    area.innerHTML = `
      <div class="number-input-wrap">
        ${hasPfx ? `<div class="number-input-prefix">${step.prefix}</div>` : ''}
        <input class="number-input ${hasPfx ? '' : 'no-prefix'}" id="number-field"
               type="number" min="${step.min||0}" max="${step.max||999999}"
               placeholder="${step.placeholder||''}" value="${answers[step.key]||''}"
               oninput="answers['${step.key}']=this.value"
               onkeydown="if(event.key==='Enter')goNext()">
        ${step.suffix ? `<div style="position:absolute;right:14px;top:50%;transform:translateY(-50%);color:var(--ink-3);font-size:13px;pointer-events:none">${step.suffix}</div>` : ''}
      </div>`
    setTimeout(() => document.getElementById('number-field')?.focus(), 80)
  }

  fireEvent('step_start', { step_index: index, step_key: step.key })
  showScreen('step')
}

function selectOption(key, value, el) {
  answers[key] = value
  el.closest('.options').querySelectorAll('.option').forEach(o => {
    o.classList.remove('selected')
    o.querySelector('.option-check').textContent = ''
  })
  el.classList.add('selected')
  el.querySelector('.option-check').textContent = '✓'
  setTimeout(() => goNext(), 280)
}

function goNext() {
  const step = steps[currentStep]
  if (step.type === 'choice' && !answers[step.key]) return
  if (step.type === 'number') {
    const v = parseFloat(answers[step.key])
    if (!v || isNaN(v)) { document.getElementById('number-field')?.focus(); return }
  }
  fireEvent('step_complete', { step_index: currentStep, step_key: step.key })
  if (currentStep < steps.length - 1) showStep(currentStep + 1)
  else showResult()
}

function goBack() {
  if (currentStep > 0) {
    fireEvent('step_back', { step_index: currentStep })
    showStep(currentStep - 1)
  }
}

// ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
//  CALCULATE
// ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
function calculateEstimate() {
  let multiplier = 1.0, sizeVal = null
  steps.forEach(s => {
    if (s.type === 'choice' && answers[s.key]) {
      const opt = s.options.find(o => o.value === answers[s.key])
      if (opt?.multiplier) multiplier *= opt.multiplier
    }
    if (s.type === 'number' && answers[s.key]) sizeVal = parseFloat(answers[s.key])
  })
  const size = sizeVal || 1500
  const base = size * 8.5 * multiplier
  const low  = Math.round(base * 0.75 / 100) * 100
  const high = Math.round(base * 1.35 / 100) * 100
  const mid  = Math.round((low + high) / 2 / 100) * 100
  return { total: mid, low, high, size, multiplier: Math.round(multiplier * 100) / 100 }
}

function showResult() {
  estimateResult = calculateEstimate()
  document.getElementById('progress-fill').style.width = '100%'
  document.getElementById('result-value').textContent  = '$' + estimateResult.total.toLocaleString()
  document.getElementById('result-range').textContent  =
    'Estimated range: $' + estimateResult.low.toLocaleString() + ' – $' + estimateResult.high.toLocaleString()

  document.getElementById('result-breakdown').innerHTML = Object.entries(answers).map(([key, val]) => {
    const step  = steps.find(s => s.key === key)
    if (!step) return ''
    const label   = step.question.replace('?','').replace(/^(What is the |What type of |How would you describe the |When are you looking to )/,'')
    const display = step.type === 'number'
      ? val + (step.suffix ? ' ' + step.suffix : '')
      : step.options?.find(o => o.value === val)?.label || val
    return `<div class="breakdown-row"><span class="breakdown-label">${label}</span><span class="breakdown-value">${display}</span></div>`
  }).join('')

  fireEvent('result_view', { estimate: estimateResult.total })
  showScreen('result')
}

// ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
//  LEAD SUBMISSION
// ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
async function submitLead() {
  const fname = document.getElementById('lead-fname').value.trim()
  const lname = document.getElementById('lead-lname').value.trim()
  const email = document.getElementById('lead-email').value.trim()
  const phone = document.getElementById('lead-phone').value.trim()
  const errEl = document.getElementById('form-error')
  errEl.classList.remove('show')

  if (!fname)                     { showFormError('Please enter your first name'); return }
  if (!email || !email.includes('@')) { showFormError('Please enter a valid email address'); return }

  const btn = document.getElementById('btn-submit')
  btn.disabled    = true
  btn.textContent = 'Submitting…'

  try {
    const leadUID = 'lead_' + Date.now().toString(36) + Math.random().toString(36).slice(2, 6)

    const { data: lead, error } = await sb.from('leads').insert({
      lead_uid:           leadUID,
      business_id:        estimator.business_id,
      estimator_id:       estimator.id,
      visitor_session_id: sessionId,
      email,
      full_name:          (fname + ' ' + lname).trim(),
      phone:              phone || null,
      answers,
      estimate_result:    estimateResult,
      crm_status:         'pending',
      status:             'new',
    }).select('id, lead_uid').single()

    if (error) throw error

    await fireEvent('lead_submit', { lead_uid: lead.lead_uid, estimate: estimateResult.total })

    if (sessionId) {
      await sb.from('visitor_sessions')
        .update({ completed: true, last_active_at: new Date().toISOString() })
        .eq('id', sessionId)
    }

    document.getElementById('success-value').textContent = '$' + estimateResult.total.toLocaleString()
    document.getElementById('success-ref').textContent   = lead.lead_uid
    document.getElementById('success-sub').textContent   =
      `Thanks ${fname}! We'll send your detailed quote to ${email} shortly.`

    showScreen('success')

  } catch(e) {
    console.error('Submit error:', e)
    showFormError('Something went wrong — please try again.')
    btn.disabled    = false
    btn.textContent = 'Send Me My Quote →'
  }
}

function showFormError(msg) {
  const el = document.getElementById('form-error')
  el.textContent = msg
  el.classList.add('show')
}

//  EVENT TRACKING
// ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
async function fireEvent(eventType, payload = {}) {
  if (!estimator?.id) return // skip if demo mode
  try {
    await sb.from('estimator_events').insert({
      estimator_id:       estimator.id,
      visitor_session_id: sessionId,
      estimator_uid:      SLUG,
      visitor_uid:        visitorUID,
      lead_uid:           payload.lead_uid || null,
      event_type:         eventType,
      step_index:         payload.step_index ?? null,
      step_key:           payload.step_key   || null,
      payload,
      occurred_at:        new Date().toISOString(),
    })
  } catch(e) {
    console.warn('Event tracking skipped:', e.message)
  }
}

  document.getElementById('screen-' + id).classList.add('active')
}
function showError(msg) {
  document.getElementById('error-message').textContent = msg
  showScreen('error')
}
